
# TESSERACT
## 1. Update system and install Tesseract + English data
sudo pacman -Syu tesseract tesseract-data-eng

## 2. Install basic build tools (needed for some python extensions)
sudo pacman -S base-devel


# PADDLE
## 1. install paddlepaddle
pip install paddlepaddle==2.6.1 -i https://mirror.baidu.com/pypi/simple

## 2. Install C++ dependencies
sudo pacman -S libglvnd openmp
